34  import com.jcabi.log.Logger;

38  import lombok.EqualsAndHashCode;
39  import lombok.ToString;
40  import org.xembly.Directive;
41  import org.xembly.Directives;

43  /**
44   * Starts Amazon ECS container, that runs Docker + SSHD on ECS. Container
45   * is configured in privileged mode. Configuration is stored in S3.
46   * See details here:
47   * @checkstyle LineLengthCheck (1 line)
48   * see <a href="http://docs.aws.amazon.com/AmazonECS/latest/developerguide/ecs-agent-config.html#ecs-config-s3">link</a>
49   * @author Yuriy Alevohin (alevohin@mail.ru)
50   * @version $Id: 5c3bde03b9677cbbe0d9844a06e9ae3a1b0674e9 $
51   * @since 2.0
52   * @todo #629 Implement com.rultor.agents.ecs.StopsECS agent. It must
53   *  stop ECS on-demand container if it was started at StartsECS agent.
54   *  StopsECS must use container ARN from /talk/ec2/[@id] to stop it.
55   * @todo #629 Add new container creation classes for StartsECS and StopsECS
56   *  to com.rultor.agents.Agents. StartsECS must be invoked before
57   *  RegistersShell agent. StopsECS must be invoked after RemovesShell agent.
58   * @todo #629 Implement the ability to use Rultor with any AWS account,
59   *  by providing the credentials to it via the encrypted assets and a
60   *  respective configuration setting in the .rultor.yml. Document the
61   *  implementation properly in the Rultor documentation.
62   */
63  @Immutable
64  @ToString
65  @EqualsAndHashCode(callSuper = false, of = { "amazon" })
66  public final class StartsECS extends AbstractAgent {
67      /**
68       * AmazonECS client provider.
69       */
70      private final transient Amazon amazon;
71  
72      /**
73       * Ctor.
74       * @param amaz Amazon
75       */
76      public StartsECS(final Amazon amaz) {
77          super("/talk[daemon and not(shell)]");
78          this.amazon = amaz;
79      }
80  
81      @Override
82      // @todo #629 Add Container params to Directive, for example publicIpAddress
83      //  Let's fix this when possible. The ticket will be in GitHub.
84      public Iterable<Directive> process(final XML xml) throws IOException {
85          final Container container = this.amazon.runOnDemand();
86          Logger.info(
87              this,
88              "ECS container %s created",
89              container
90          );
91          return new Directives().xpath("/talk")
92              .add("ec2")
93              .attr("id", container.getContainerArn());
94      }
95  }
