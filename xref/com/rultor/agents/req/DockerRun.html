50  /**
51   * Docker run command.
52   *
53   * @author Yegor Bugayenko (yegor256@gmail.com)
54   * @version $Id: e22301d8bfc703057dfed992d8b6bc8104dfeb2b $
55   * @since 1.0
56   * @checkstyle MultipleStringLiteralsCheck (500 lines)
57   * @checkstyle ClassDataAbstractionCouplingCheck (500 lines)
58   */
59  @Immutable
60  @ToString
61  @EqualsAndHashCode(of = { "profile", "xpath" })
62  @SuppressWarnings("PMD.TooManyMethods")
63  final class DockerRun {
64  
65      /**
66       * Profile.
67       */
68      private final transient Profile profile;
69  
70      /**
71       * Node name.
72       */
73      private final transient String xpath;
74  
75      /**
76       * Ctor.
77       * @param prof Profile
78       * @param path Node name in profile XML, XPath
79       */
80      DockerRun(final Profile prof, final String path) {
81          this.profile = prof;
82          this.xpath = path;
83      }
84  
85      /**
86       * Make a script to run.
87       * @return Script
88       * @throws IOException If fails
89       */
90      public Iterable<String> script() throws IOException {
91          final Iterable<String> trap;
92          if (this.profile.read().nodes("/p/entry[@key='uninstall']").isEmpty()) {
93              trap = Collections.emptyList();
94          } else {
95              trap = new Joined<String>(
96                  new SolidList<String>("function", "clean_up()", "{"),
97                  DockerRun.scripts(
98                      this.profile.read(), "/p/entry[@key='uninstall']"
99                  ),
100                 new SolidList<String>("}", ";"),
101                 new SolidList<String>("trap", "clean_up", "EXIT", ";")
102             );
103         }
104         return new Joined<String>(
105             trap,
106             DockerRun.scripts(
107                 this.profile.read(), "/p/entry[@key='install']"
108             ),
109             DockerRun.scripts(this.node(), "entry[@key='script']")
110         );
111     }
112 
113     /**
114      * Make a list of env vars for docker.
115      * @param extra Extra vars
116      * @return Envs
117      * @throws IOException If fails
118      */
119     public Iterable<String> envs(final Map<String, String> extra)
120         throws IOException {
121         final List<String> entries = new LinkedList<>();
122         for (final Entry<String, String> ent : extra.entrySet()) {
123             entries.add(
124                 String.format(
125                     "%s=%s", ent.getKey(), ent.getValue()
126                 )
127             );
128         }
129         return new Joined<String>(
130             DockerRun.envs(this.profile.read(), "/p/entry[@key='env']"),
131             DockerRun.envs(this.node(), "entry[@key='env']"),
132             new SolidList<String>(entries)
133         );
134     }
135 
136     /**
137      * Get xpath.
138      * @return XML
139      * @throws IOException If fails
140      */
141     private XML node() throws IOException {
142         final XML node;
143         final Collection<XML> nodes = this.profile.read().nodes(this.xpath);
144         if (nodes.isEmpty()) {
145             node = new XMLDocument("<x/>");
146         } else {
147             node = nodes.iterator().next();
148         }
149         return node;
150     }
151 
152     /**
153      * Get items from XML.
154      * @param xml The XML
155      * @param path The XPath
156      * @return Items
157      */
158     private static Iterable<String> scripts(final XML xml, final String path) {
159         final Collection<String> items = new LinkedList<String>();
160         if (!xml.nodes(path).isEmpty()) {
161             final XML node = xml.nodes(path).get(0);
162             if (node.nodes("item").isEmpty()) {
163                 items.addAll(DockerRun.lines(node));
164             } else {
165                 for (final String cmd : node.xpath("item/text()")) {
166                     items.add(cmd.trim());
167                 }
168             }
169         }
170         final Collection<String> scripts = new LinkedList<String>();
171         for (final String item : items) {
172             scripts.add(neutralize(item));
173             scripts.add(";");
174         }
175         return scripts;
176     }
177 
178     /**
179      * Is hash character inside double or single quotes.
180      * @param item String to check.
181      * @param pos Position of the hash.
182      * @return If hash is in quotes.
183      */
184     private static boolean inquotes(final String item, final int pos) {
185         return StringUtils.countMatches(item.substring(0, pos), "\"") % 2 == 1
186             || StringUtils.countMatches(item.substring(0, pos), "'") % 2 == 1;
187     }
188 
189     /**
190      * Neutralize comment contained in the script line.
191      * @param item Script element.
192      * @return Script element with invisible comment
193      */
194     private static String neutralize(final String item) {
195         final int start = item.indexOf('#');
196         final String result;
197         if (start == 0 || start > 0 && item.charAt(start - 1) != '\\'
198             && !DockerRun.inquotes(item, start)) {
199             result = new StringBuilder(item.substring(0, start))
200                 .append('`')
201                 .append(item.substring(start))
202                 .append('`')
203                 .toString();
204         } else {
205             result = item;
206         }
207         return result;
208     }
209 
210     /**
211      * Environments from XML by XPath.
212      * @param xml The XML
213      * @param path The XPath
214      * @return Items
215      */
216     private static Iterable<String> envs(final XML xml, final String path) {
217         final Collection<String> envs = new LinkedList<>();
218         if (!xml.nodes(path).isEmpty()) {
219             final XML node = xml.nodes(path).get(0);
220             final Collection<String> parts;
221             if (node.nodes("item").iterator().hasNext()) {
222                 parts = node.xpath("item/text()");
223             } else if (node.nodes("entry").iterator().hasNext()) {
224                 parts = new LinkedList<>();
225                 for (final XML env : node.nodes("./entry")) {
226                     parts.add(
227                         String.format(
228                             "%s=%s", env.xpath("@key").get(0),
229                             env.xpath("text()").get(0)
230                         )
231                     );
232                 }
233             } else {
234                 parts = DockerRun.lines(node);
235             }
236             envs.addAll(
237                 new SolidList<String>(
238                     new org.cactoos.collection.Mapped<>(
239                         input -> String.format("%s", input),
240                         parts
241                     )
242                 )
243             );
244         }
245         return envs;
246     }
247 
248     /**
249      * Get lines from a single XML node.
250      * @param node Node to get text() from
251      * @return Lines found
252      */
253     private static Collection<String> lines(final XML node) {
254         final Collection<String> lines = new LinkedList<String>();
255         if (node.node().hasChildNodes()) {
256             lines.addAll(
257                 new SolidList<String>(
258                     new org.cactoos.collection.Mapped<>(
259                         input -> input.trim(),
260                         Arrays.asList(
261                             StringUtils.split(node.xpath("text()").get(0), '\n')
262                         )
263                     )
264                 )
265             );
266         }
267         return lines;
268     }
